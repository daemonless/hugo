## Deployment

### Podman Compose

```yaml
services:
  hugo:
    image: ghcr.io/daemonless/hugo:latest
    container_name: hugo
    environment:
      - PUID=1000
      - PGID=1000
      - TZ=UTC
      - HUGO_BASEURL=http://localhost:1313
    volumes:
      - /path/to/containers/hugo/app:/app
    ports:
      - 1313:1313
    restart: unless-stopped
```
